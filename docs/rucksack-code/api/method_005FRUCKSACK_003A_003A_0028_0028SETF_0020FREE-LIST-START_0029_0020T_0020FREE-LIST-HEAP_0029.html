(defmethod (setf free-list-start) (pointer (heap free-list-heap)
                                           &optional (size-class 0))
  (setf (pointer-value (free-list-pointer size-class) heap) pointer
        ;; Keep copy in memory
        (aref (slot-value heap 'starts) size-class) pointer))
