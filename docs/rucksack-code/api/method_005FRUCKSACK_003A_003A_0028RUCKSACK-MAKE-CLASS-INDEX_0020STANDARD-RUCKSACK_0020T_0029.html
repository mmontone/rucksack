(defmethod rucksack-make-class-index 
           ((rucksack standard-rucksack) class
            &key (index-spec '(btree :key< < :value= p-eql)))
  ;; A class index maps object ids to objects.
  (declare (ignore class))
  (make-index index-spec t))
