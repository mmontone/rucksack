(defmethod add-schema ((table schema-table) (schema schema))
  (setf (gethash (schema-id schema) (schema-table-by-id table))
        schema)
  (push schema
        (gethash (schema-class-name schema) (schema-table-by-name table) '()))
  (setf (dirty-p table) t))
