(defmethod print-object ((cache standard-cache) stream)
  (print-unreadable-object (cache stream :type t :identity nil)
    (format stream "of size ~D, heap ~S and ~D objects in memory."
            (cache-size cache)
            (pathname (heap-stream (heap cache)))
            (cache-count cache))))
