(defmethod deserialize-contents ((marker (eql +p-object+)) stream)
  ;; Return a proxy.  The proxy contents will be automatically loaded
  ;; when necessary.
  (let ((id (deserialize stream)))
    (make-instance 'proxy
                   :object-id id)))
