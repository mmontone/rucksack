(defun rucksack-list-slot-indexes (rucksack)
  (let ((result '()))
    (rucksack-map-slot-indexes rucksack
                               (lambda (class-name slot-name slot-index)
                                 (declare (ignore slot-index))
                                 (push (cons class-name slot-name)
                                       result)))
    result))
